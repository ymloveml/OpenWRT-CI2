# ========== 平台选择 ==========
# 支持的品牌：CMIOT GLINET LINKSYS QIHOO REDMI XIAOMI ZN JDCLOUD
CONFIG_TARGET_qualcommax=y                  # 选择高通平台
CONFIG_TARGET_qualcommax_ipq60xx=y          # 选择IPQ60xx芯片系列
CONFIG_TARGET_MULTI_PROFILE=y               # 支持多配置文件构建
CONFIG_TARGET_PER_DEVICE_ROOTFS=y           # 为每个设备生成单独的根文件系统

# ========== 内核配置 ==========
CONFIG_KERNEL_ARM_PMU=y                     # 启用ARM性能监控单元
CONFIG_KERNEL_CFQ_GROUP_IOSCHED=y           # 启用CFQ组IO调度器
CONFIG_KERNEL_CGROUP_DEVICE=y               # 启用设备cgroup控制
CONFIG_KERNEL_CGROUP_FREEZER=y              # 启用cgroup任务冻结功能
CONFIG_KERNEL_CGROUP_HUGETLB=y              # 启用大页内存cgroup控制
CONFIG_KERNEL_CGROUP_NET_PRIO=y             # 启用网络优先级cgroup控制
CONFIG_KERNEL_CGROUP_PERF=y                 # 启用性能监控cgroup控制
CONFIG_KERNEL_HUGETLBFS=y                   # 启用大页文件系统
CONFIG_KERNEL_HUGETLB_PAGE=y                # 启用大页内存支持
CONFIG_KERNEL_MEMCG_SWAP_ENABLED=y          # 启用内存cgroup交换支持
CONFIG_KERNEL_NET_CLS_CGROUP=y              # 启用网络分类cgroup支持
CONFIG_KERNEL_PERF_EVENTS=y                 # 启用性能事件子系统
# CONFIG_KERNEL_PREEMPT_RCU is not set     # 禁用可抢占RCU（实时内核选项）

# ========== NSS配置 ==========
# NSS (Network Subsystem Services) 是高通网络加速子系统
CONFIG_FEED_nss_packages=n                  # 禁用NSS包源（可能已集成到主源码）
CONFIG_FEED_sqm_scripts_nss=n               # 禁用SQM脚本NSS包源

# 内存配置
CONFIG_IPQ_MEM_PROFILE_1024=n               # 禁用1024MB内存配置
CONFIG_IPQ_MEM_PROFILE_512=y                # 启用512MB内存配置（适合IPQ6010等设备）
CONFIG_ATH11K_MEM_PROFILE_1G=n              # 禁用WiFi 1GB内存配置
CONFIG_ATH11K_MEM_PROFILE_512M=y            # 启用WiFi 512MB内存配置

# NSS功能组件
CONFIG_PACKAGE_sqm-scripts-nss=y            # 启用NSS加速的SQM脚本（智能队列管理）
CONFIG_PACKAGE_kmod-qca-nss-crypto=y        # 启用NSS硬件加密加速模块
