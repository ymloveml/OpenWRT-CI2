# =====================================
# 瑞芯微(Rockchip)平台配置文件
# 此文件定义了瑞芯微ARMv8架构设备的配置
# =====================================

# ========== 支持的品牌 ==========
# NANOPC NANOPI FASTRHINO ORANGEPI

# ========== 平台选择 ==========
CONFIG_TARGET_rockchip=y                    # 选择瑞芯微平台
CONFIG_TARGET_rockchip_armv8=y              # 选择ARMv8架构（64位）

# ========== 设备列表 ==========
# 以下是所有启用的瑞芯微平台设备
# FriendlyARM系列产品
CONFIG_TARGET_DEVICE_rockchip_armv8_DEVICE_friendlyarm_nanopc-t4=y           # NanoPC-T4
CONFIG_TARGET_DEVICE_rockchip_armv8_DEVICE_friendlyarm_nanopc-t6=y           # NanoPC-T6
CONFIG_TARGET_DEVICE_rockchip_armv8_DEVICE_friendlyarm_nanopi-r2c-plus=y     # NanoPi R2C Plus
CONFIG_TARGET_DEVICE_rockchip_armv8_DEVICE_friendlyarm_nanopi-r2c=y          # NanoPi R2C
CONFIG_TARGET_DEVICE_rockchip_armv8_DEVICE_friendlyarm_nanopi-r2s=y          # NanoPi R2S
CONFIG_TARGET_DEVICE_rockchip_armv8_DEVICE_friendlyarm_nanopi-r3s=y          # NanoPi R3S
CONFIG_TARGET_DEVICE_rockchip_armv8_DEVICE_friendlyarm_nanopi-r4s-enterprise=y # NanoPi R4S企业版
CONFIG_TARGET_DEVICE_rockchip_armv8_DEVICE_friendlyarm_nanopi-r4s=y          # NanoPi R4S
CONFIG_TARGET_DEVICE_rockchip_armv8_DEVICE_friendlyarm_nanopi-r4se=y         # NanoPi R4SE
CONFIG_TARGET_DEVICE_rockchip_armv8_DEVICE_friendlyarm_nanopi-r5c=y          # NanoPi R5C
CONFIG_TARGET_DEVICE_rockchip_armv8_DEVICE_friendlyarm_nanopi-r5s=y          # NanoPi R5S
CONFIG_TARGET_DEVICE_rockchip_armv8_DEVICE_friendlyarm_nanopi-r6c=y          # NanoPi R6C
CONFIG_TARGET_DEVICE_rockchip_armv8_DEVICE_friendlyarm_nanopi-r6s=y          # NanoPi R6S

# FastRhino系列产品
CONFIG_TARGET_DEVICE_rockchip_armv8_DEVICE_lunzn_fastrhino-r66s=y            # FastRhino R66S
CONFIG_TARGET_DEVICE_rockchip_armv8_DEVICE_lunzn_fastrhino-r68s=y            # FastRhino R68S

# OrangePi系列产品
CONFIG_TARGET_DEVICE_rockchip_armv8_DEVICE_xunlong_orangepi-5-plus=y         # Orange Pi 5 Plus
CONFIG_TARGET_DEVICE_rockchip_armv8_DEVICE_xunlong_orangepi-5=y              # Orange Pi 5
CONFIG_TARGET_DEVICE_rockchip_armv8_DEVICE_xunlong_orangepi-r1-plus-lts=y    # Orange Pi R1 Plus LTS
CONFIG_TARGET_DEVICE_rockchip_armv8_DEVICE_xunlong_orangepi-r1-plus=y        # Orange Pi R1 Plus

# ========== WiFi驱动配置 ==========
# 联发科WiFi芯片驱动
CONFIG_PACKAGE_kmod-mt7915e=y               # MT7915 WiFi 6 PCIe驱动
CONFIG_PACKAGE_kmod-mt7921e=y               # MT7921 WiFi 6E PCIe驱动
CONFIG_PACKAGE_kmod-mt7925e=y               # MT7925 WiFi 7 PCIe驱动
CONFIG_PACKAGE_kmod-mt7921-common=y         # MT7921系列通用驱动
CONFIG_PACKAGE_kmod-mt792x-common=y         # MT792x系列通用驱动

# WiFi固件
CONFIG_PACKAGE_kmod-mt7915-firmware=y       # MT7915固件
CONFIG_PACKAGE_kmod-mt7916-firmware=y       # MT7916固件
CONFIG_PACKAGE_kmod-mt7921-firmware=y       # MT7921固件
CONFIG_PACKAGE_kmod-mt7922-firmware=y       # MT7922固件
CONFIG_PACKAGE_kmod-mt7925-firmware=y       # MT7925固件
CONFIG_PACKAGE_wpad-openssl=y               # 支持OpenSSL的WPA/WPA2/WPA3认证

# ========== GPU驱动 ==========
CONFIG_PACKAGE_kmod-drm-panfrost=y          # Panfrost GPU驱动（用于Mali GPU）
CONFIG_PACKAGE_kmod-drm-rockchip=y          # 瑞芯微DRM驱动

# ========== 分区和文件系统配置 ==========
CONFIG_TARGET_KERNEL_PARTSIZE=128           # 内核分区大小为128MB
CONFIG_TARGET_ROOTFS_PARTSIZE=1024          # 根文件系统分区大小为1024MB (1GB)
CONFIG_TARGET_ROOTFS_EXT4FS=n               # 禁用EXT4根文件系统（可能使用其他文件系统）

# ========== 插件调整 ==========
CONFIG_PACKAGE_luci-app-dockerman=y         # 启用Docker管理Web界面
